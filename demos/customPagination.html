function pagItemClick(i)
{
  $("#owl-demo").trigger("owl.goTo", i);
}

$(document).ready(function() {

  var owl = $("#owl-demo");

  owl.owlCarousel({
    autpPlay: true,
    singleItem: true,
    navigation : false,
    pagination : false,
    afterAction : afterAction
  });

  function afterAction(){
    this.owl.owlItems.each
    (
      function(i){
        $("#pagitem" + i).css(
          "opacity" , "0.3"
          );
      }
    );

    $("#pagitem" + this.owl.currentItem).css("opacity", "1");
  }
});
